SELECT
E.EMPLOYEE_ID 사번, E.FIRST_NAME 이름, E.LAST_NAME 성, NVL(D.DEPARTMENT_NAME,'<NOT-ASSIGNED>') 부서명, E.HIRE_DATE 입사년
FROM EMPLOYEES E, DEPARTMENTS D
WHERE E.DEPARTMENT_ID = D.DEPARTMENT_ID(+)AND HIRE_DATE BETWEEN '99/01/01' AND '99/12/31';
